485 interrupt void mainISR(void)
486 {
487 
488  static uint16_t stCnt = 0;
489 
490  // toggle status LED
491  if(gLEDcnt++ > (uint_least32_t)(USER_ISR_FREQ_Hz / LED_BLINK_FREQ_Hz))
492  {
493  HAL_toggleLed(halHandle,(GPIO_Number_e)HAL_Gpio_LED2);
494  gLEDcnt = 0;
495  }
496 
497 
498  // acknowledge the ADC interrupt
499  HAL_acqAdcInt(halHandle,ADC_IntNumber_1);
500 
501 
502  // convert the ADC data
503  HAL_readAdcData(halHandle,&gAdcData);
504 
505 
506  // Run the SpinTAC Components
507  if(stCnt++ >= ISR_TICKS_PER_SPINTAC_TICK) {
508  ST_runVelPlanTick(stHandle);
509  ST_runVelPlan(stHandle, ctrlHandle);
510  ST_runVelMove(stHandle, ctrlHandle);
511  ST_runVelCtl(stHandle, ctrlHandle);
512  stCnt = 1;
513  }
514 
515 
516  // run the controller
517  CTRL_run(ctrlHandle,halHandle,&gAdcData,&gPwmData);
518 
519 
520  // write the PWM compare values
521  HAL_writePwmData(halHandle,&gPwmData);
522 
523 
524  // setup the controller
525  CTRL_setup(ctrlHandle);
526 
527 
528  return;
529 } // end of mainISR() function
