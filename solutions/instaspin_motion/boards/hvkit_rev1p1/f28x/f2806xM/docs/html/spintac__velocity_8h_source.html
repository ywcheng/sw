216 inline ST_Handle ST_init(void *pMemory, const size_t numBytes)
217 {
218  ST_Handle handle;
219  ST_Obj *obj;
220 
221  handle = (ST_Handle)pMemory; // assign the handle
222  obj = (ST_Obj *)handle; // assign the object
223 
224  // init the ST VelId object
225  obj->velIdHandle = STVELID_init(&obj->vel.id, sizeof(ST_VelId_t));
226  // init the ST VelCtl object
227  obj->velCtlHandle = STVELCTL_init(&obj->vel.ctl, sizeof(ST_VelCtl_t));
228  // init the ST VelMove object
229  obj->velMoveHandle = STVELMOVE_init(&obj->vel.move, sizeof(ST_VelMove_t));
230  // init the ST VelPlan object
231  obj->velPlanHandle = STVELPLAN_init(&obj->vel.plan, sizeof(ST_VelPlan_t));
232  // init the ST PosConv object
233  obj->posConvHandle = STPOSCONV_init(&obj->vel.conv, sizeof(ST_PosConv_t));
234  // get the ST Version object
235  obj->versionHandle = ST_initVersion(&obj->version, sizeof(ST_Ver_t));
236 
237  return handle;
238 }

242
