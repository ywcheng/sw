310

313 
314

316  if(gMotorVars.UserErrorCode != USER_ErrorCode_NoError)

318  for(;;)

427  // Waiting for enable system flag to be set

429 
430  // Dis-able the Library internal PI. Iq has no reference now

546 
547  }

631 #ifdef DRV8301_SPI
