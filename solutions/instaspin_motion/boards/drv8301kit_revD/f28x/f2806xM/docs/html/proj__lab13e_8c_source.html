478 interrupt void mainISR(void)
479 {
480 
481  static uint16_t stCnt = 0;
482  CTRL_Obj *obj = (CTRL_Obj *)ctrlHandle;
483 
484  // toggle status LED
485  if(gLEDcnt++ > (uint_least32_t)(USER_ISR_FREQ_Hz / LED_BLINK_FREQ_Hz))
486  {
487  HAL_toggleLed(halHandle,(GPIO_Number_e)HAL_Gpio_LED2);
488  gLEDcnt = 0;
489  }
490 
491 
492  // compute the electrical angle
493  ENC_calcElecAngle(encHandle, HAL_getQepPosnCounts(halHandle));
494 
495 
496  // acknowledge the ADC interrupt
497  HAL_acqAdcInt(halHandle,ADC_IntNumber_1);
498 
499 
500  // convert the ADC data
501  HAL_readAdcData(halHandle,&gAdcData);
502 
503 
504  // Run the SpinTAC Components
505  if(stCnt++ >= ISR_TICKS_PER_SPINTAC_TICK) {
506  ST_runPosConv(stHandle, encHandle, ctrlHandle);
507  ST_runPosMove(stHandle);
508  ST_runPosCtl(stHandle, ctrlHandle);
509  stCnt = 1;
510  }
511 
512 
513  if(USER_MOTOR_TYPE == MOTOR_Type_Induction) {
514  // update the electrical angle for the SLIP module
515  SLIP_setElectricalAngle(slipHandle, ENC_getElecAngle(encHandle));
516  // compute the amount of slip
517  SLIP_run(slipHandle);
518 
519 
520  // run the controller
521  CTRL_run(ctrlHandle,halHandle,&gAdcData,&gPwmData,SLIP_getMagneticAngle(slipHandle));
522  }
523  else {
524  // run the controller
525  CTRL_run(ctrlHandle,halHandle,&gAdcData,&gPwmData,ENC_getElecAngle(encHandle));
526  }
527 
528  // write the PWM compare values
529  HAL_writePwmData(halHandle,&gPwmData);
530 
531 
532  // setup the controller
533  CTRL_setup(ctrlHandle);
534 
535  // if we are forcing alignment, using the Rs Recalculation, align the eQEP angle with the rotor angle
536  if((EST_getState(obj->estHandle) == EST_State_Rs) && (USER_MOTOR_TYPE == MOTOR_Type_Pm))
537  {
538  ENC_setZeroOffset(encHandle, (uint32_t)(HAL_getQepPosnMaximum(halHandle) - HAL_getQepPosnCounts(halHandle)));
539  }
540 
541  return;
542 } // end of mainISR() function

545 void updateGlobalVariables_motor(CTRL_Handle handle, ST_Handle sthandle)
546 {
547  uint32_t ProTime_tick, ProTime_mtick;
548  CTRL_Obj *obj = (CTRL_Obj *)handle;
549  ST_Obj *stObj = (ST_Obj *)sthandle;
550 
551  // get the speed estimate
552  gMotorVars.Speed_krpm = _IQmpy(STPOSCONV_getVelocityFiltered(stObj->posConvHandle), _IQ(ST_SPEED_KRPM_PER_PU));
553 
554  // get the position error
555  gMotorVars.PositionError_MRev = STPOSCTL_getPositionError_mrev(stObj->posCtlHandle);
556 
557  // get the torque estimate
558  gMotorVars.Torque_Nm = USER_computeTorque_Nm(handle, gTorque_Flux_Iq_pu_to_Nm_sf, gTorque_Ls_Id_Iq_pu_to_Nm_sf);
559 
560  // get the magnetizing current
561  gMotorVars.MagnCurr_A = EST_getIdRated(obj->estHandle);
562 
563  // get the rotor resistance
564  gMotorVars.Rr_Ohm = EST_getRr_Ohm(obj->estHandle);
565 
566  // get the stator resistance
567  gMotorVars.Rs_Ohm = EST_getRs_Ohm(obj->estHandle);
568 
569  // get the stator inductance in the direct coordinate direction
570  gMotorVars.Lsd_H = EST_getLs_d_H(obj->estHandle);
571 
572  // get the stator inductance in the quadrature coordinate direction
573  gMotorVars.Lsq_H = EST_getLs_q_H(obj->estHandle);
574 
575  // get the flux in V/Hz in floating point
576  gMotorVars.Flux_VpHz = EST_getFlux_VpHz(obj->estHandle);
577 
578  // get the flux in Wb in fixed point
579  gMotorVars.Flux_Wb = USER_computeFlux(handle, gFlux_pu_to_Wb_sf);
580 
581  // get the controller state
582  gMotorVars.CtrlState = CTRL_getState(handle);
583 
584  // get the estimator state
585  gMotorVars.EstState = EST_getState(obj->estHandle);
586 
587  // Get the DC buss voltage
588  gMotorVars.VdcBus_kV = _IQmpy(gAdcData.dcBus,_IQ(USER_IQ_FULL_SCALE_VOLTAGE_V/1000.0));
589 
590  // get the Iq reference from the position controller
591  gMotorVars.IqRef_A = _IQmpy(STPOSCTL_getTorqueReference(stObj->posCtlHandle), _IQ(USER_IQ_FULL_SCALE_CURRENT_A));
592 
593  // gets the Position Controller status
594  gMotorVars.SpinTAC.PosCtlStatus = STPOSCTL_getStatus(stObj->posCtlHandle);
595 
596  // get the inertia setting
597  gMotorVars.SpinTAC.InertiaEstimate_Aperkrpm = _IQmpy(STPOSCTL_getInertia(stObj->posCtlHandle), _IQ(ST_SPEED_PU_PER_KRPM * USER_IQ_FULL_SCALE_CURRENT_A));
598 
599  // get the friction setting
600  gMotorVars.SpinTAC.FrictionEstimate_Aperkrpm = _IQmpy(STPOSCTL_getFriction(stObj->posCtlHandle), _IQ(ST_SPEED_PU_PER_KRPM * USER_IQ_FULL_SCALE_CURRENT_A));
601 
602  // get the Position Controller error
603  gMotorVars.SpinTAC.PosCtlErrorID = STPOSCTL_getErrorID(stObj->posCtlHandle);
604 
605  // get the Position Move status
606  gMotorVars.SpinTAC.PosMoveStatus = STPOSMOVE_getStatus(stObj->posMoveHandle);
607 
608  // get the Position Move profile time
609  STPOSMOVE_getProfileTime_tick(stObj->posMoveHandle, &ProTime_tick, &ProTime_mtick);
610  gMotorVars.SpinTAC.PosMoveTime_ticks = ProTime_tick;
611  gMotorVars.SpinTAC.PosMoveTime_mticks = ProTime_mtick;
612 
613  // get the Position Move error
614  gMotorVars.SpinTAC.PosMoveErrorID = STPOSMOVE_getErrorID(stObj->posMoveHandle);
615 
616  // get the Position Converter error
617  gMotorVars.SpinTAC.PosConvErrorID = STPOSCONV_getErrorID(stObj->posConvHandle);
618 
619  return;
620 } // end of updateGlobalVariables_motor() function
