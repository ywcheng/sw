223 inline ST_Handle ST_init(void *pMemory, const size_t numBytes)
224 {
225  ST_Handle handle;
226  ST_Obj *obj;
227 
228  handle = (ST_Handle)pMemory; // assign the handle
229  obj = (ST_Obj *)handle; // assign the object
230 
231  // init the ST PosCtl object
232  obj->posCtlHandle = STPOSCTL_init(&obj->pos.ctl, sizeof(ST_PosCtl_t));
233  // init the ST PosMove object
234  obj->posMoveHandle = STPOSMOVE_init(&obj->pos.move, sizeof(ST_PosMove_t));
235  // init the ST PosPlan object
236  obj->posPlanHandle = STPOSPLAN_init(&obj->pos.plan, sizeof(ST_PosPlan_t));
237  // init the ST PosConv object
238  obj->posConvHandle = STPOSCONV_init(&obj->pos.conv, sizeof(ST_PosConv_t));
239  // get the ST Version object
240  obj->versionHandle = ST_initVersion(&obj->version, sizeof(ST_Ver_t));
241 
242  return handle;
243 }

260  }
261  STPOSCONV_setEnable(obj->posConvHandle, true);
